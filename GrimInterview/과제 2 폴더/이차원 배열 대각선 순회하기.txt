#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
   int answer = 0;
    if (board.empty()) return 0;
    size_t boardRow = board.size();
    size_t boardCol = board[0].size();
    for (size_t i = 0; i < boardRow; ++i) {
        for (size_t j = 0; j < boardCol; ++j) {
            if (i + j <= k)
                answer += board[i][j];
        }
    }
    return answer;
}
